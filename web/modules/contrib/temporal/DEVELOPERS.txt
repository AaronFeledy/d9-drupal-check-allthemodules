Provided below are some code examples of using the services provided

// TemporalListService Testing
/** @var $listService \Drupal\temporal\TemporalListService */
$listService = \Drupal::service('temporal.list');

// Entity methods
$query = $listService->prepareFieldValuesByEntityId('node', 'article', 'field_monthly_hours', 4);
$query = $listService->prepareFieldValuesByEntityIdList('node', 'article', 'field_monthly_hours', array(2,3,4));
$query = $listService->prepareFieldValuesByField('node', 'article', 'field_monthly_hours');
$query = $listService->prepareFieldValuesByBundle('node', 'article');
$query = $listService->prepareFieldValuesByEntityType('node');

// Temporal Type methods
$query = $listService->prepareFieldValuesByTemporalTypeEntityId('hours', 4);
$query = $listService->prepareFieldValuesByTemporalTypeEntityIdList('hours', array(2,3,4));
$query = $listService->prepareFieldValuesByTemporalType('hours');

$results = $listService->getResults($query);

dpm($results);

// TemporalRangeService Testing
$start_date = mktime(0,0,0,6,1,2011);
$end_date = mktime(0,0,0,6,30,2016);

$entity_list = [
  2,
];

/* @var $rangeService \Drupal\temporal\TemporalRangeService */
$rangeService = \Drupal::service('temporal.range');
// Temporal Type methods

$query = $rangeService->rangeFieldValuesByTemporalTypeEntityIdList($start_date, $end_date, 'hours', $entity_list);
$query = $rangeService->rangeFieldValuesByTemporalType($start_date, $end_date, 'hours');

$results = $rangeService->getResults($query);

dpm($results);
